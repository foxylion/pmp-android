395            }
396            
397            IPCProvider.getInstance().startUpdate();
398            try {
399                // remember that illegal service features have to be enabled once their missing PS get installed
400                for (App app : this.cache.getApps().values()) {
401                    boolean appChanged = false;
402                    
403                    for (ServiceFeature sf : this.cache.getServiceFeatures().get(app).values()) {
404                        if (!sf.isAvailable()) {
405                            sf.forceRecache();
406                            
407                            // if the service feature was only missing this RG, rollout the changes
408                            if (sf.isAvailable()) {
409                                appChanged = true;
410                            }
411                        }
412                    }
413                    
414                    if (appChanged) {
415                        app.verifyServiceFeatures();
416                    }
417                }
418                
419                // remember that illegal presets have to be enabled once their missing PS get installed
420                for (Preset p : this.cache.getAllPresets()) {
421                    if (!p.isAvailable()) {
422                        p.forceRecache();
423                        
424                        // if the preset was only missing this RG, rollout the changes
425                        if (p.isAvailable()) {
426                            p.rollout();
427                        }
428                    }
429                }
430                
431            } finally {
432                IPCProvider.getInstance().endUpdate();
433            }
434            
435            FileLog.get().logWithForward(this, null, FileLog.GRANULARITY_COMPONENT_CHANGES, Level.CONFIG,
436                    "ResourceGroup '%s' has been successfully installed.", rgPackage);
437            return true;
438        } catch (ParserException xmlpe) {
439            /* error during XML validation */
440            FileLog.get().logWithForward(this, xmlpe, FileLog.GRANULARITY_COMPONENT_CHANGES, Level.WARNING,
441                    "ResourceGroup '%s' has failed registration with PMP: Could not verify XML file.", rgPackage);
442            return false;
443        }
