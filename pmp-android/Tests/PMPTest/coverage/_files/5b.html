424        
425        ResourceGroup rg = this.cache.getResourceGroups().get(rgPackage);
426        if (rg == null) {
427            return false;
428        } else {

430            rg.delete();
431            // delete the class files / apk / etc
432            PluginProvider.getInstance().uninstall(rgPackage);
433            this.cache.getResourceGroups().remove(rgPackage);
434            
435            IPCProvider.getInstance().startUpdate();
436            try {
437                // remember that service features have to be disabled once their required PS get uninstalled
438                for (App app : this.cache.getApps().values()) {
439                    boolean appChanged = false;
440                    
441                    for (ServiceFeature sf : this.cache.getServiceFeatures().get(app).values()) {
442                        if (sf.isAvailable()) {
443                            sf.forceRecache();
444                            
445                            // if the service feature will be missing this RG, rollout the changes
446                            if (!sf.isAvailable()) {
447                                appChanged = true;
448                            }
449                        }
450                    }
451                    
452                    if (appChanged) {
453                        app.verifyServiceFeatures();
454                    }
455                }
456                
457                // remember that presets have to be disabled once their required PS get uninstalled
458                for (Preset preset : this.cache.getAllPresets()) {
459                    if (preset.isAvailable() && !preset.isDeleted()) {
460                        preset.forceRecache();
461                        
462                        // if the preset will be missing this RG, rollout the changes
463                        if (!preset.isAvailable()) {
464                            preset.rollout();
465                        }
466                    }
467                }
468            } finally {
469                IPCProvider.getInstance().endUpdate();
470            }
471            
472            return true;
